# Commands to deploy Voila App on Heroku

mkdir voila-simple-app
cd voila-simple-app
git init
git remote add origin https://github.com/<UserName>/<RepoName>.github
git pull origin master

python -m venv venv
venv/bin/activate
pip install jupyter voila
jupyter notebook

voila app.ipynb

pip freeze > requirements.txt
nano runtime.txt
nano Procfile

git add .
git commit -m "initial commits for simple voila app to Heroku"
git push origin master

heroku login
heroku create
git push heroku master
heroku open
heroku logs --tail
